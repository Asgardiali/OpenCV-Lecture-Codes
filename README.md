# main.py
import stack
path = 'Resources/Shapes.jpg'
img = cv2.imread(path)
imgGray = cv2.cvtColor(img,cv2.COLOR_BGR2GRAY)
imgStack = stack.stackImages(0.5, ([img, imgGray]))
cv2.imshow("Image with gray level",imgStack)
